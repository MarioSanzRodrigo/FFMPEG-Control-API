69  int proc_instance_index, uint32_t fifo_ctx_maxsize[PROC_IO_NUM],
70  log_ctx_t *log_ctx, va_list arg)

72  int i, ret_code, end_code= STAT_ERROR;
73  proc_ctx_t *proc_ctx= NULL;
74  fifo_elem_alloc_fxn_t fifo_elem_alloc_fxn= {0};
75  LOG_CTX_INIT(log_ctx);
76 
77  /* Check arguments */
78  CHECK_DO(proc_if!= NULL, return NULL);
79  CHECK_DO(settings_str!= NULL, return NULL);
80  CHECK_DO(fifo_ctx_maxsize!= NULL , return NULL);
81  // Note: 'log_ctx' is allowed to be NULL
82 
83  /* Check mandatory call-backs existence */
84  CHECK_DO(proc_if->open!= NULL, goto end);
85  CHECK_DO(proc_if->close!= NULL, goto end);
86  CHECK_DO(proc_if->process_frame!= NULL, goto end);
87 
88  /* Open (allocate) the specific processor (PROC) instance */
89  proc_ctx= proc_if->open(proc_if, settings_str, LOG_CTX_GET(), arg);
90  CHECK_DO(proc_ctx!= NULL, goto end);
91 
92  /* Set processor (PROC) interface context structure */
93  proc_ctx->proc_if= proc_if;
94 
95  /* Set PROC register index. */
96  proc_ctx->proc_instance_index= proc_instance_index;
97 
98  /* API mutual exclusion lock */
99  ret_code= pthread_mutex_init(&proc_ctx->api_mutex, NULL);
100  CHECK_DO(ret_code== 0, goto end);
101 
102  /* Set external LOG module */
103  proc_ctx->log_ctx= LOG_CTX_GET();
104 
105  /* Initialize input FIFO buffer */
106  fifo_elem_alloc_fxn.elem_ctx_dup=
107  proc_if->iput_fifo_elem_opaque_dup!= NULL?
108  (fifo_elem_ctx_dup_fxn_t*)proc_if->iput_fifo_elem_opaque_dup:
109  (fifo_elem_ctx_dup_fxn_t*)proc_frame_ctx_dup;
110  fifo_elem_alloc_fxn.elem_ctx_release=
111  proc_if->iput_fifo_elem_opaque_release!= NULL?
112  (fifo_elem_ctx_release_fxn_t*)
113  proc_if->iput_fifo_elem_opaque_release:
114  (fifo_elem_ctx_release_fxn_t*)proc_frame_ctx_release;
115 
116  proc_ctx->fifo_ctx_array[PROC_IPUT]= fifo_open(
117  fifo_ctx_maxsize[PROC_IPUT], 0, &fifo_elem_alloc_fxn);
118  CHECK_DO(proc_ctx->fifo_ctx_array[PROC_IPUT]!= NULL, goto end);
119 
120  /* Initialize output FIFO buffer */
121  fifo_elem_alloc_fxn.elem_ctx_dup=
122  proc_if->oput_fifo_elem_opaque_dup!= NULL?
123  (fifo_elem_ctx_dup_fxn_t*)proc_if->oput_fifo_elem_opaque_dup:
124  (fifo_elem_ctx_dup_fxn_t*)proc_frame_ctx_dup;
125  fifo_elem_alloc_fxn.elem_ctx_release=
126  (fifo_elem_ctx_release_fxn_t*)proc_frame_ctx_release;
127  proc_ctx->fifo_ctx_array[PROC_OPUT]= fifo_open(
128  fifo_ctx_maxsize[PROC_OPUT], 0, &fifo_elem_alloc_fxn);
129  CHECK_DO(proc_ctx->fifo_ctx_array[PROC_OPUT]!= NULL, goto end);
130 
131  /* Initialize input/output fair-locks */
132  for(i= 0; i< PROC_IO_NUM; i++) {
133  fair_lock_t *fair_lock= fair_lock_open();
134  CHECK_DO(fair_lock!= NULL, goto end);
135  proc_ctx->fair_lock_io_array[i]= fair_lock;
136  }
137 
138  /* At last, launch PROC thread */
139  proc_ctx->flag_exit= 0;
140  proc_ctx->start_routine= (const void*(*)(void*))proc_thr;
141  ret_code= pthread_create(&proc_ctx->proc_thread, NULL, proc_thr, proc_ctx);
142  CHECK_DO(ret_code== 0, goto end);
143 
144  end_code= STAT_SUCCESS;
145 end:
146  if(end_code!= STAT_SUCCESS)
147  proc_if->close(&proc_ctx);
148  return proc_ctx;

151 void proc_close(proc_ctx_t **ref_proc_ctx)
152 {
153  proc_ctx_t *proc_ctx= NULL;
154  LOG_CTX_INIT(NULL);
155  LOGD(">>%s\n", __FUNCTION__); //comment-me
156 
157  if(ref_proc_ctx== NULL)
158  return;
159 
160  if((proc_ctx= *ref_proc_ctx)!= NULL) {
161  const proc_if_t *proc_if;
162  void *thread_end_code= NULL;
163  LOG_CTX_SET(proc_ctx->log_ctx);
164 
165  /* Join processing thread first
166  * - set flag to notify we are exiting processing;
167  * - unlock input and output FIFO's;
168  * - join thread.
169  */
170  proc_ctx->flag_exit= 1;
171  fifo_set_blocking_mode(proc_ctx->fifo_ctx_array[PROC_IPUT], 0);
172  fifo_set_blocking_mode(proc_ctx->fifo_ctx_array[PROC_OPUT], 0);
173  LOGD("Waiting thread to join... "); // comment-me
174  pthread_join(proc_ctx->proc_thread, &thread_end_code);
175  if(thread_end_code!= NULL) {
176  ASSERT(*((int*)thread_end_code)== STAT_SUCCESS);
177  free(thread_end_code);
178  thread_end_code= NULL;
179  }
180  LOGD("joined O.K.\n"); // comment-me
181 
182  /* Release API mutual exclusion lock */
183  ASSERT(pthread_mutex_destroy(&proc_ctx->api_mutex)== 0);
184 
185  /* Release input and output FIFO's */
186  fifo_close(&proc_ctx->fifo_ctx_array[PROC_IPUT]);
187  fifo_close(&proc_ctx->fifo_ctx_array[PROC_OPUT]);
188 
189  /* Release input/output fair locks */
190  fair_lock_close(&proc_ctx->fair_lock_io_array[PROC_IPUT]);
191  fair_lock_close(&proc_ctx->fair_lock_io_array[PROC_OPUT]);
192 
193  /* Close the specific PROC instance */
194  CHECK_DO((proc_if= proc_ctx->proc_if)!= NULL, return); // sanity check
195  CHECK_DO(proc_if->close!= NULL, return); // sanity check
196  proc_if->close(ref_proc_ctx);
197  }
198  LOGD("<<%s\n", __FUNCTION__); //comment-me
199 }

314 static void* proc_thr(void *t)
315 {
316  const proc_if_t *proc_if;
317  proc_ctx_t* proc_ctx= (proc_ctx_t*)t;
318  int ret_code, *ref_end_code= NULL;
319  int (*process_frame)(proc_ctx_t*, fifo_ctx_t*, fifo_ctx_t*)= NULL;
320  fifo_ctx_t *iput_fifo_ctx= NULL, *oput_fifo_ctx= NULL;
321  LOG_CTX_INIT(NULL);
322 
323  /* Allocate return context; initialize to a default 'ERROR' value */
324  ref_end_code= (int*)malloc(sizeof(int));
325  CHECK_DO(ref_end_code!= NULL, return NULL);
326  *ref_end_code= STAT_ERROR;
327 
328  /* Check arguments */
329  CHECK_DO(proc_ctx!= NULL, goto end);
330 
331  LOG_CTX_SET(proc_ctx->log_ctx);
332 
333  /* Get PROC processing callback */
334  proc_if= proc_ctx->proc_if;
335  CHECK_DO(proc_if!= NULL, goto end);
336  process_frame= proc_if->process_frame;
337  CHECK_DO(process_frame!= NULL, goto end);
338 
339  /* Get input/output FIFO buffers */
340  iput_fifo_ctx= proc_ctx->fifo_ctx_array[PROC_IPUT];
341  oput_fifo_ctx= proc_ctx->fifo_ctx_array[PROC_OPUT];
342  CHECK_DO(iput_fifo_ctx!= NULL && oput_fifo_ctx!= NULL, goto end);
343 
344  /* Run processing thread */
345  while(proc_ctx->flag_exit== 0) {
346  ret_code= process_frame(proc_ctx, iput_fifo_ctx, oput_fifo_ctx);
347  if(ret_code== STAT_EOF)
348  proc_ctx->flag_exit= 1;
349  else if(ret_code!= STAT_SUCCESS)
350  schedule(); // Avoid CPU-consuming closed loops
351  }
352 
353  *ref_end_code= STAT_SUCCESS;
354 end:
355  return (void*)ref_end_code;
356 }
